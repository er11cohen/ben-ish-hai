מצב גרסת הדף

הדף הזה לא נבדק

הלכות שנה ראשונה – פרשת לך-לך – המשך הלכות ציצית

פתיחה

"אֲנִי אֵל שַׁדַּי הִתְהַלֵּךְ לְפָנַי וֶהְיֵה תָמִים", נִרְאֶה לִי בְּסִיַּעְתָּא דִּשְׁמַיָּא דְּיָדוּעַ שֵׁם "אֵל שַׁדַּי" מֵאִיר בְּעוֹלָם הַבְּרִיאָה, וְהַטַּעַם כִּי עוֹלָם הַבְּרִיאָה הוּא עוֹלָם הַשְּׁלִישִׁי שֶׁשָּׁם קְדֻשָּׁה מְשֻׁלֶּשֶׁת, וְלָכֵן מֵאִיר שָׁם אֵל שַׁדַּי דְּחֶשְׁבּוֹנוֹ מְשֻׁלָּשׁ, כָּל שְׁלִישׁ הוּא קט"ו כְּמִנְיַן "חֶסֶד גָּדוֹל", וְלָכֵן מֹשֶׁה רַבֵּנוּ עָלָיו הַשָּׁלוֹם שֶׁזָּכָה אֶל הַנְּשָׁמוֹת דְּבִינָה דִּבְרִיאָה כְּמוֹ שֶׁכָּתוּב בְּשַׁעַר הַפְּסוּקִים בְּפָרָשַׁת בְּרֵאשִׁית, נִקְרָא מֹשֶׁה שֶׁהוּא מִסְפַּר "אֵל שַׁדַּי", שֶׁכָּל שְׁלִישׁ הוּא מִסְפַּר "חֶסֶד גָּדוֹל" שֶׁהוּא הָיָה בְּסוֹד הַחֲסָדִים בְּנֶפֶשׁ רוּחַ וּנְשָׁמָה שֶׁלּוֹ, כְּמוֹ שֶׁכָּתוּב "וַתֵּרֶא אֹתוֹ כִּי טוֹב הוּא", וְלָכֵן נִתְּנָה תּוֹרָה עַל יָדוֹ דְּהַתּוֹרָה נָתְנָה לָנוּ מֵעוֹלָם הַבְּרִיאָה.

וְהִנֵּה יָדוּעַ שֶׁיִּשְׂרָאֵל נִשְׁמָתָם חֲצוּבָה מִתַּחַת כִּסֵּא הַכָּבוֹד שֶׁהוּא עוֹלָם הַבְּרִיאָה, בְּסוֹד בָּרוּךְ אֱלֹהֵינוּ שֶׁבְּרָאָנוּ לִכְבוֹדוֹ, שֶׁשָּׁם מֵאִיר שֵׁם "אֵל שַׁדַּי" שֶׁהוּא מְשֻׁלָּשׁ, כָּל שְׁלִישׁ מִסְפַּר "חֶסֶד גָּדוֹל", וְלָכֵן אִם יִשְׂרָאֵל יִהְיוּ שְׁלֵמִים בַּעֲבוֹדָה בְּאַהֲבָה מְשֻׁלֶּשֶׁת, שֶׁיִּהְיֶה לָהֶם בַּעֲבוֹדַת הַשֵּׁם יִתְבָּרַךְ אַהֲבָה בְּכַוָּנָה, אַהֲבָה בְּדִבּוּר, אַהֲבָה בְּמַעֲשֶׂה -- אָז יָאִיר בָּהֶם שֵׁם "אֵל שַׁדַּי" שֶׁהוּא חֶשְׁבּוֹן מְשֻׁלָּשׁ, כָּל שְׁלִישׁ מִסְפַּר "חֶסֶד גָּדוֹל", כִּי הָאַהֲבָה הִיא מִצַּד הַחֶסֶד בְּסוֹד "זֶרַע אַבְרָהָם אוֹהֲבִי", וְגַם שָׁם הוּא סוֹד הַגְּדוֹלָה, לְכָךְ יָאִיר בָּהֶם "אֵל שַׁדַּי" שֶׁהוּא ג' פְּעָמִים "חֶסֶד גָּדוֹל" ג' פְּעָמִים "חֶסֶד גָּדוֹל" שווה 345 בְּנֶפֶשׁ רוּחַ נְשָׁמָה שֶׁלָּהֶם, וְעַל שִׁלּוּשׁ זֶה רָמַז לָהֶם בַּמֶּה שֶׁאָמַר "וִהְיִיתֶם לִי סְגֻלָּה", דֻּגְמַת הַסֶגוֹ"ל שֶׁהוּא נְקֻדַּת הַחֶסֶד וְהוּא מְשֻׁלָּשׁ בְּשָׁלֹשׁ נְקֻדּוֹת.

וְיָדוּעַ הוּא דִּשְׁלֵמוֹת הָעֲבוֹדָה בִּשְׁלֹשָׁה פְּעָמִים אַהֲבָה הַנִּזְכָּרִים, רְמוּזָה בָּאוֹתִיּוֹת שֶׁלִּפְנֵי שֵׁם הֲוָיָ"ה שֶׁהֵם כוז"ו שֶׁמִּסְפָּרָם ג' פְּעָמִים "אַהֲבָה" ג' פְּעָמִים "אַהֲבָה" שווה 39, וּבָזֶה פֵּרַשְׁתִּי הַפָּסוּק "אַחֲרֵי ה' אֱלֹהֵיכֶם תֵּלֵכוּ", כִּי כוז"ו שֶׁהֵם מִסְפַּר שָׁלֹשׁ פְּעָמִים "אַהֲבָה", אַחֲרֵי אוֹתִיּוֹת שֵׁם הֲוָיָ"ה בְּאַלְפָ"א בֵּיתָ"א.

וּבָזֶה יוּבַן בְּסִיַּעְתָּא דִּשְׁמַיָּא הָרֶמֶז כָּאן "אֲנִי אֵל שַׁדַּי", שֶׁהוּא שֵׁם הַמֵּאִיר בִּבְרִיאָה שֶׁהוּא מְשֻׁלָּשׁ בְּשָׁלֹשׁ פְּעָמִים "חֶסֶד גָּדוֹל", לָכֵן "הִתְהַלֵּךְ לִפְנֵי" כְּלוֹמַר בְּאוֹתִיּוֹת שֶׁהֵם לִפְנֵי שְׁמִי שֶׁהֵם מִסְפַּר שָׁלֹשׁ פְּעָמִים אַהֲבָה, שֶׁהוּא בְּסוֹד הַחֶסֶד הַגָּדוֹל, וּבָזֶה תִּהְיֶה תָּמִים בְּהֶאָרַת "אֵל שַׁדַּי".

וּמֵאַחַר שֶׁאֲנַחְנוּ צְרִיכִין לַעֲבֹד אֶת הַשֵּׁם יִתְבָּרַךְ וּלְקַיֵּם מִצְוֹתָיו בִּשְׁלֹשָׁה פְּעָמִים אַהֲבָה שֶׁהֵם בְּכַוָּנָה וּבְדִבּוּר וּבְמַעֲשֶׂה, לָכֵן הַצִּיצִית שֶׁהוּא זֵכֶר לְכָל מִצְוֹת ה' עוֹשִׂין בּוֹ טַ"ל כְּרִיכוֹת, לִרְמֹז שֶׁאֲנַחְנוּ צְרִיכִין לִרְאוֹתוֹ לִזְכֹּר בּוֹ מִצְוַת ה' בְּאַהֲבָה מְשֻׁלֶּשֶׁת בְּכַוָּנָה וּבְדִבּוּר וּבְמַעֲשֶׂה, וּלְכָךְ נִקְרָא "טַלִּית" שֶׁהוּא טַ"ל ת"י, וְהַיְנוּ טַ"ל רֶמֶז לְטַ"ל הַנִּזְכָּר, וְת"י הוּא מִסְפַּר קָדוֹשׁ -- שֶׁיִּמָּשֵׁךְ לָנוּ קְדֻשָּׁה מִמִּצְוָה זוֹ שֶׁל טַ"ל כְּרִיכוֹת הַצִּיצִיּוֹת, כִּי אֵין לָנוּ הַשָּׂגָה אֶלָּא בַּצִּיצִיּוֹת שֶׁבַּטַּלִּית וְלֹא בְּגוּף הַטַּלִּית עַצְמוֹ:

הלכות

א

הַגְּדִיל יַעֲשֵׂהוּ חֻלְיוֹת חֻלְיוֹת, כָּל חֻלְיָה מִשְּׁלֹשָׁה כְּרִיכוֹת, וּבַתְּחִלָּה יִקְשֹׁר שְׁנֵי קְשָׁרִים סָמוּךְ לַכָּנָף, וְיִכְרֹךְ שֶׁבַע כְּרִיכוֹת, וְיִקְשֹׁר שְׁנֵי קְשָׁרִים אֲחֵרִים, וְיִכְרֹךְ שְׁמוֹנָה כְּרִיכוֹת, וְיִקְשֹׁר שְׁנֵי קְשָׁרִים אֲחֵרִים, וְיִכְרֹךְ אַחַת עֶשְׂרֵה כְּרִיכוֹת, וְיִקְשֹׁר שְׁנֵי קְשָׁרִים, וְיִכְרֹךְ שְׁלֹשׁ עֶשְׂרֵה כְּרִיכוֹת, וְאַחַר כָּךְ יִקְשֹׁר עוֹד שְׁנֵי קְשָׁרִים, וַהֲרֵי סַךְ הַכֹּל טַ"ל כְּרִיכוֹת כְּמִנְיַן "הֲוָיָ"ה אֶחָד", וּכְמוֹ שֶׁכָּתַב רַבֵּנוּ זַ"ל בְּסֵפֶר הַכַּוָּנוֹת, וְעָשִׂיתִי סִימָן לַדָּבָר זֶה "תֵּן חֵלֶק לְשִׁבְעָה, וְגַם לִשְׁמוֹנָה", נִמְצָא תְּחִלָּה שִׁבְעָה כְּרִיכוֹת, וְאַחַר כָּךְ שְׁמוֹנָה כְּרִיכוֹת, וְאַחַר כָּךְ תַּעֲשֶׂה אַחַת עֶשְׂרֵה כְּמִנְיַן אוֹתִיּוֹת "לְשִׁבְעָה לִשְׁמוֹנָה" שֶׁהֵם אַחַת עֶשְׂרֵה אוֹתִיּוֹת, וְאַחַר כָּךְ תַּעֲשֶׂה שְׁלֹשׁ עֶשְׂרֵה כְּמִנְיַן אוֹתִיּוֹת "תֶּן חֵלֶק לְשִׁבְעָה וְגַם" שֶׁהֵם שְׁלֹשׁ עֶשְׂרֵה אוֹתִיּוֹת:

ב

גּוֹי שֶׁתָּלָה הַחוּטִין שֶׁל צִיצִית בַּטַּלִּית, אוֹ שֶׁעָשָׂה הַחֻלְיוֹת וְהַקְּשָׁרִים פְּסוּלָה, אֲבָל הָאִשָּׁה כְּשֵׁרָה לְהַטִּיל הַצִּיצִית בַּטַּלִּית וְלַעֲשׂוֹת הַחֻלְיוֹת וְהַקְּשָׁרִים, וְיֵשׁ מַחְמִירִין שֶׁלֹּא לַעֲשׂוֹת עַל יְדֵי נָשִׁים, וְלָכֵן לְכַתְּחִלָּה צָרִיךְ לְהִזָּהֵר שֶׁלֹּא לַעֲשׂוֹת תְּלִיַּת הַחוּטִין וּכְרִיכָתָן וּקְשִׁירָתָן עַל יְדֵי נָשִׁים אֶלָּא עַל יְדֵי אֲנָשִׁים, אֲבָל טְוִיָּה וּשְׁזִירָה עוֹשִׂין גַּם לְכַתְּחִלָּה עַל יְדֵי הַנָּשִׁים, וְהַקָּטָן דִּינוֹ כְּאִשָּׁה לְעִנְיָן זֶה, וְלָכֵן לְכַתְּחִלָּה אֵין לַעֲשׂוֹת תְּלִיַּת הַחוּטִין וְהַכְּרִיכוֹת וְהַקְּשָׁרִים עַל יְדֵי קָטָן אֶלָּא עַל יְדֵי גָּדוֹל, וּמִיהוּ כָּל שֶׁהוּא בֶּן שְׁלֹשׁ עֶשְׂרֵה שָׁנָה וְיוֹם אֶחָד יַעֲשֶׂה לְכַתְּחִלָּה, וְאֵין צָרִיךְ לִבְדֹּק אִם יֵשׁ לוֹ שְׁתֵּי שְׂעָרוֹת:

ג

אִם הֵטִיל יִשְׂרָאֵל צִיצִית בַּבֶּגֶד שֶׁלֹּא בְּכַוָּנָה, כְּלוֹמַר שֶׁלֹּא נִתְכַּוֵּן לְשֵׁם מִצְוַת צִיצִית יֵשׁ פּוֹסְלִין וְיֵשׁ מַכְשִׁירִין וַהֲלָכָה כִּסְבָרַת הַפּוֹסְלִין, וְלָכֵן צָרִיךְ לְהַתִּירָם וְיַחֲזֹר וְיִתְלֵם לְשֵׁם מִצְוַת צִיצִית, וְאִם הוּא שַׁבָּת שֶׁאִי אֶפְשָׁר לוֹ לְהַתִּירָם וּלְקָשְׁרָם, וְאֵין לוֹ טַלִּית אַחֶרֶת כְּשֵׁרָה יֵשׁ לִסְמֹךְ עַל הַמַּתִּירִים בְּאוֹתוֹ הַיּוֹם דַּוְקָא, וְאַחַר שַׁבָּת יְתַקְּנֶנָּה אַךְ לֹא יְבָרֵךְ דְּסָפֵק בְּרָכוֹת לְהָקֵל וּבְרָכוֹת אֵינָם מְעַכְּבוֹת, וְלָכֵן כָּל אָדָם יִזָּהֵר שֶׁלֹּא לִסְמֹךְ בִּתְלִיַּת הַחוּטִין וְתִקּוּנָם אֶלָּא עַל אָדָם יְרֵא שָׁמַיִם וּמֻחְזָק בְּכַשְׁרוּת וּבָקִי וְיוֹדֵעַ לְכַוֵּן לְשֵׁם מִצְוָה וְלַעֲשׂוֹת כָּל הַדְּבָרִים כְּתִקּוּנָם, וְעַל כֵּן יִזָּהֵר לוֹמַר בְּפִיו מִקֹּדֶם "הֲרֵינִי מַטִּיל צִיצִיּוֹת אֵלּוּ לְשֵׁם מִצְוַת צִיצִית", יַעַן דְּיֵשׁ אוֹמְרִים כָּל דָּבָר הַצָּרִיךְ כַּוָּנָה לִשְׁמָהּ צָרִיךְ לְהוֹצִיא הַכַּוָּנָה בִּשְׂפָתָיו. וּכְמוֹ שֶׁכָּתְבוּ הָאַחֲרוֹנִים בְּדִין הַטְּוִיָּה וּשְׁזִירָה, וְטוֹב לוֹמַר: לְשֵׁם יִחוּד קָדְשָׁא בְּרִיךְ הוּא וּשְׁכִינְתֵּהּ וְכוּ' הֲרֵנִי מוּכָן לְהַטִּיל צִיצִיּוֹת בְּאַרְבַּע כַּנְפוֹת הַבֶּגֶד הַזֶּה כְּמוֹ שֶׁצִּוָּנוּ ה' אֱלֹהֵינוּ בְּתוֹרָתוֹ "וְעָשׂוּ לָהֶם צִיצִת עַל כַּנְפֵי בִגְדֵיהֶם", וְנֶאֱמַר "גְּדִלִים תַּעֲשֶׂה לָּךְ עַל אַרְבַּע כַּנְפוֹת כְּסוּתְךָ אֲשֶׁר תְּכַסֶּה בָּהּ", וִיהִי רָצוֹן מִלְּפָנֶיךָ ה' אֱלֹהֵינוּ וֵאלֹהֵי אֲבוֹתֵינוּ כְּשֵׁם שֶׁאֲנַחְנוּ מְקַיְּמִים מִצְוָה זוֹ בְּחוּטֵי לָבָן כָּךְ תְּזַכֵּנוּ לְקַיֵּם אוֹתָהּ כְּתִקְנָהּ בְּחוּטֵי תְּכֵלֶת כַּכָּתוּב בַּתּוֹרָה "וְנָתְנוּ עַל צִיצִת הַכָּנָף פְּתִיל תְּכֵלֶת" וַיְהִי נֹעַם וְכוּ':

ד

טוֹב לְהִזָּהֵר שֶׁלֹּא לַחְתֹּךְ חוּטֵי הַצִּיצִית בְּסַכִּין, עַל דֶּרֶךְ שֶׁנֶּאֱמַר "לֹא תָנִיף עֲלֵיהֶם בַּרְזֶל" אֶלָּא יַחְתְּכֵם בְּשִׁנָּיו כְּשֶׁצָּרִיךְ לְחָתְכָם:

ה

אִם שָׁאַל טַלִּית מְצֻיֶּצֶת אִם שָׁאַל אוֹתָהּ לְלָבְשָׁהּ כְּדֵי לַעֲלוֹת לַדּוּכָן אוֹ לַעֲלוֹת לְסֵפֶר תּוֹרָה וְכַיּוֹצֵא הֲרֵי זֶה לֹא יְבָרֵךְ עָלָיו, אֲבָל אִם שָׁאַל בַּעֲבוּר לְקַיֵּם מִצְוַת עִטּוּף בַּצִּיצִית, הִנֵּה אִם שְׁאָלוֹ מִן הַשַּׁמָּשׁ שֶׁלֹּא מִדַּעַת בְּעָלִים לֹא יְבָרֵךְ עָלָיו דִּבְרָכוֹת אֵינָם מְעַכְּבוֹת וְכֵיוָן דְּאִיכָּא פְּלוּגְתָּא סָפֵק בְּרָכוֹת לְהָקֵל, אֲבָל אִם שְׁאָלוֹ מִדַּעַת בְּעָלִים לְהִתְעַטֵּף בּוֹ לְקַיֵּם מִצְוַת צִיצִית יוּכַל לְבָרֵךְ, דְּאָמְרִינַן מִסְּתָמָא נְתָנוֹ לוֹ בְּמַתָּנָה עַל מְנָת לְהַחֲזִיר וַהֲרֵי הוּא שֶׁלּוֹ, וְאַף עַל גַּב דְּגַם בָּזֶה יֵשׁ אוֹמְרִים דְּאֵין לְבָרֵךְ עַל טַלִּית שְׁאוּלָה כְּלָל, מִכָּל מָקוֹם בִּכְהַאי גַּוְנָא יֵשׁ לִסְמֹךְ עַל הַמִּנְהָג וְאֵין לִמְחוֹת בְּיַד הַמְּבָרֵךְ, וּמִיהוּ יְרֵא שָׁמַיִם יִתְרַחֵק מִטַּלִּית שְׁאוּלָה הֵיכָא דְאֶפְשָׁר לוֹ, וְטַלִּיתוֹת שֶׁל הֶקְדֵּשׁ שֶׁמֻּנָּחִים בְּבֵית הַכְּנֶסֶת בְּתוֹרַת צְדָקָה לְלָבְשָׁם אוֹתָם שֶׁאֵין לָהֶם טַלִּיתוֹת לְכֻלֵּי עָלְמָא יְבָרֵךְ עֲלֵיהֶם, וַאֲפִלּוּ אִם לוֹבְשָׁן כְּדֵי לַעֲלוֹת לְסֵפֶר תּוֹרָה וּכְמוֹ שֶׁכָּתְבוּ הָאַחֲרוֹנִים זַ"ל בָּזֶה:

ו

בַּזְּמַן הַזֶּה חָזֵינַן רֻבָּא דְּעָלְמָא קָפְדִי שֶׁלֹּא יִלְבְּשׁוּ אֲחֵרִים צִיצִית וּתְפִלִּין שֶׁלָּהֶם, וְלָכֵן אֵין לִקַּח מִן הַשַּׁמָּשׁ צִיצִית וּתְפִלִּין שֶׁל אֲחֵרִים שֶׁלֹּא מִדַּעַת בְּעָלִים, אֶלָּא אִם כֵּן יוֹדֵעַ הַשַּׁמָּשׁ בְּבֵרוּר שֶׁהַבְּעָלִים לָא קָפְדִי, וְאֵין כָּל הַדֵּעוֹת שָׁווֹת בִּדְבָרִים אֵלּוּ הֲלֹא גַּם אָנֹכִי מַכִּיר בְּעַצְמִי דַּאֲנָא מֵאִינָשֵׁי דְּקָפְדִי בָּזֶה וְהַטְּעָמִים שׁוֹנִים:

ז

מֻתָּר לְהַתִּיר הַצִּיצִיּוֹת מִן הַטַּלִּית בֵּין גָּדוֹל בֵּין קָטָן בִּשְׁבִיל הִדּוּר מִצְוָה לָשׂוּם בִּמְקוֹמָן צִיצִיּוֹת חֲדָשִׁים וְנָאִים יוֹתֵר, אַף עַל פִּי דְּאֵלּוּ הַצִּיצִיּוֹת שֶׁנּוֹטְלָם הוּא מַשְׁלִיכָם וְגוֹנְזָם וְאֵין מַנִּיחָם בְּטַלִּית אַחֶרֶת שָׁרִי מִשּׁוּם נוֹי מִצְוָה, וְכֵן אִם נִפְסַק כָּנָף אֶחָד וְרוֹצֶה לְחַדֵּשׁ אַרְבַּעְתָּם בִּשְׁבִיל נוֹי שֶׁיִּהְיוּ שָׁוִים וְדוֹמִין זֶה לָזֶה, מֻתָּר לְהָסִיר כָּל הַשְּׁלֹשָׁה אֲחֵרוֹת וְלַעֲשׂוֹת הָאַרְבַּע כְּנָפוֹת חֲדָשִׁים שֶׁיִּהְיוּ שָׁוִים יַחַד בִּשְׁבִיל נוֹי מִצְוָה:

ח

אַף עַל פִּי שֶׁהַכְּרִיכוֹת שֶׁל הַצִּיצִית עֲשָׂאָם כְּמִסְפַּר שֵׁם הֲוָיָ"ה, כִּי יֵשׁ שֶׁמִּנְהָגָם בְּכָךְ וְאֵין נוֹהֲגִין בִּכְרִיכוֹת שֶׁל טַ"ל הֲרֵי זֶה מֻתָּר לְהַתִּירָם, וְאֵין בָּזֶה חֲשַׁשׁ מוֹחֵק אֶת הַשֵּׁם כְּמוֹ שֶׁכָּתְבוּ הָאַחֲרוֹנִים זַ"ל:

ט

כְּשֶׁרוֹצֶה לְהַתִּיר הַצִּיצִית כְּדֵי לְהַנִּיחַ יוֹתֵר יָפִים וְקָשֶׁה עָלָיו הַטִּרְחָא לְהַתִּירָן מֻתָּר לְחָתְכָם אַף עַל פִּי שֶׁהֵם נִפְסָדִים בָּזֶה לְגַמְרֵי, מִיהוּ יֵשׁ מַחְמִירִין בָּזֶה וְאוֹסְרִים לְחָתְכָם וְצָרִיךְ לָחוּשׁ לְדִבְרֵיהֶם, וְרַק בִּשְׁעַת הַדְּחָק וְצֹרֶךְ גָּדוֹל יֵשׁ לִסְמֹךְ עַל הַמַּתִּירִים כָּךְ הֶעֱלוּ הָאַחֲרוֹנִים זַ"ל:

י

טַלִּית מְצֻיֶּצֶת שֶׁנֶּחְלְקָה לִשְׁנֵי חֲלָקִים אָסוּר לַחֲזֹר לְתָפְרָהּ עִם הַצִּיצִיּוֹת שֶׁבָּהּ מִשּׁוּם תַּעֲשֶׂה וְלֹא מִן הֶעָשׂוּי, כֵּיצַד יַעֲשֶׂה יַתִּיר הַצִּיצִיּוֹת שֶׁל חֵלֶק אֶחָד מֵהֶם דַּוְקָא, וְאַחַר כָּךְ יִתְפֹּר הַחֲלָקִים יַחַד וְנִמְצָא הָוֵי זֶה הַחֵלֶק שֶׁאֵין בּוֹ צִיצִית כְּמוֹ תּוֹסֶפֶת עַל חֵלֶק הָרִאשׁוֹן, וְאַחַר כָּךְ יַעֲשֶׂה צִיצִית מֵחָדָשׁ בַּחֵלֶק הַזֶּה שֶׁהוּא תּוֹסֶפֶת לָרִאשׁוֹן, כֵּן הַעֲלֵתִי בְּסִפְרֵי הַק' "מִקַּבְצְאֵל" מִסִּפְרִי הָאַחֲרוֹנִים זַ"ל:

יא

שִׁעוּר הַטַּלִּית גָּדוֹל הוּא כְּדֵי שֶׁיַּקִּיף כָּל רֹאשׁוֹ וְכָל גּוּפוֹ עַד הֶחָזֶה פָּנִים וְאָחוֹר כְּמוֹ שֶׁכָּתַב רַבֵּינוּ הָאֲרִ"י זַ"ל בְּסֵפֶר הַכַּוָּנוֹת, וַחֲכָמֵינוּ זִכְרוֹנָם לִבְרָכָה אָמְרוּ כְּדֵי שֶׁיְּכַסֶּה רֹאשׁוֹ וְרֻבּוֹ שֶׁל קָטָן, וּבְסִפְרֵי הַק' "מִקַּבְצְאֵל" הֵבֵאתִי דִּבְרֵי רַבֵּינוּ הָאֲרִ"י זַ"ל מַה שֶּׁכָּתַב בְּסֵפֶר הַכַּוָּנוֹת וּמַה שֶׁכָּתוּב בְּ"עוֹלַת תָּמִיד" וּמַה שֶׁכָּתוּב בְּ"נְהַר שָׁלוֹם" לְרַבֵּינוּ הָרָשָׁ"שׁ זַ"ל, וְהִשְׁוֵיתִי דִּבְרֵי רַבֵּינוּ הָאֲרִ"י זַ"ל עִם דִּבְרֵי חֲכָמֵינוּ זִכְרוֹנָם לִבְרָכָה כִּי שִׁעוּר זֶה דְּנָקְטִי חֲכָמֵינוּ זִכְרוֹנָם לִבְרָכָה בְּקָטָן דְּבָעֵי רֹב גּוּפוֹ הוּא שָׁוֶה עִם שִׁעוּר דְּנָקֵט רַבֵּינוּ הָאֲרִ"י זַ"ל בְּגָדוֹל עַד הֶחָזֶה וְלָא נָקֵט רֹב גּוּפוֹ, וְשָׁם נִסְתַּפַּקְתִּי אִם זֶה הַשִּׁעוּר דְּבָעֵינַן בַּגָּדוֹל עַד הֶחָזֶה אִם אָדָם אָרֹךְ הַרְבֵּה יוּכַל לִלְבֹּשׁ טַלִּית הַמַּגִּיעַ בְּאָדָם בֵּינוֹנִי עַד הֶחָזֶה, אַף עַל פִּי שֶׁאֵילָיו אֵינוֹ מַגִּיעַ עַד הֶחָזֶה, וְהַעֲלֵתִי דְּאָזְלִינַן לְחוּמְרָא וּבְעִנְיָן שֶׁיַּגִּיעַ לָזֶה הֶעָרוּךְ עַד הֶחָזֶה, כִּי מִסְּתִימוּת דִּבְרֵי רַבֵּינוּ הָאֲרִ"י זַ"ל נִרְאֶה יוֹתֵר שֶׁצָּרִיךְ לְשַׁעֵר כָּל אֶחָד כְּפִי מִדָּתוֹ, מִיהוּ לְקֻלָּא לָא עָבְדִינַן שֶׁאִם יֵשׁ קְצַר קוֹמָה הַרְבֵּה שֶׁהוּא גָּדוֹל בְּשָׁנִים וְרַק קוֹמָתוֹ קְצָרָה כְּמִדַּת קָטָן בֶּן ט' שָׁנִים אֵינוֹ יָכוֹל לִלְבֹּשׁ טַלִּית שֶׁיַּגִּיעַ אֵלָיו עַד הֶחָזֶה כֵּיוָן שֶׁזֶּה הַטַּלִּית אֵינוֹ מַגִּיעַ בַּגָּדוֹל עַד הֶחָזֶה:

יב

בְּשִׁעוּר טַלִּית קָטָן שֶׁלּוֹבְשִׁין תַּחַת הַבְּגָדִים, בְּסִפְרֵי הַק' "מִקַּבְצְאֵל" הֵבֵאתִי דִּבְרֵי הָאַחֲרוֹנִים זַ"ל בָּזֶה וְשָׁם הֵבֵאתִי מַה שֶׁהֶעֱלָה הָרַב הַגָּאוֹן רַבִּי חַיִּים פָלָאגִ'י זַ"ל בְּלֵב חַיִּים סִימָן צ"ט דְּהַנָּכוֹן לָצֵאת יְדֵי חוֹבָתוֹ לְכֻלֵּי עָלְמָא לְשַׁעֵר בְּטַלִּית קָטָן שֶׁיִּהְיֶה אַמָּה וָחֵצִי שֶׁהֵם ל"ו גּוּדָלִים בְּאֹרֶךְ, וְאַמָּה וָחֵצִי בְּרֹחַב. וּמָה גַּם לִסְבָרַת רַבֵּינוּ הָאֲרִ"י זַ"ל דְּצָרִיךְ לְכַסּוֹת גּוּפוֹ פָּנִים וְאָחוֹר עַד כָּאן דְּבָרָיו עַיִן שָׁם, וְהָרוֹאֶה יִרְאֶה בְּדִבְרֵי רַבֵּינוּ הָאֲרִ"י זַ"ל שֶׁלֹּא אָמַר עַל טַלִּית קָטָן שֶׁיְּכַסֶּה פָּנִים וְאָחוֹר, אֶלָּא אָמַר שִׁעוּר שֶׁיְּעַטֵּף אֶת הַגּוּף מִלְּבַד הַזְּרוֹעוֹת, וּבָזֶה נִרְאֶה דְּאֵין צָרִיךְ שֶׁיִּהְיֶה הָרֹחַב אַמָּה וָחֵצִי שֶׁהוּא ל"ו גּוּדָלִים, אַךְ בְּאֹרֶךְ מוֹדֵינָא לֵיהּ דְּשַׁפִּיר הוֹרָה כֵּן לִסְבָרַת הַטּוּר וּדְעִמֵּהּ דְּצָרִיךְ לַעֲשׂוֹת אָרְכּוֹ אַמָּה וָחֵצִי שֶׁהוּא ל"ו גּוּדָלִים, וְהִנֵּה פֹּה עִירֵנוּ בַּגְדָאד יע"א מִנְהָגָם לַעֲשׂוֹת הַטַּלִּית קָטָן אָרְכּוֹ אַמָּה וּרְבִיעַ בְּאַמַּת הַמְּדִינָה, שֶׁעוֹלֶה הָאֹרֶךְ בְּזֶה יוֹתֵר מֵאַרְבָּעִים גּוּדָלִים, אַךְ הָרֹחַב עוֹשִׂים אוֹתוֹ שְׁלֹשָׁה רְבִיעֵי אַמָּה שֶׁעוֹלֶה יוֹתֵר מכ"ד גּוּדָלִים וּבִתְשׁוּבָה לְעִיר אַחֶרֶת הוֹרֵיתִי לַשּׁוֹאֵל לַעֲשׂוֹת הָרֹחַב כ"ד גּוּדָלִים וְהָאֹרֶךְ ל"ו גּוּדָלִים:

יג

נָשִׁים וַעֲבָדִים פְּטוּרִים מִמִּצְוַת צִיצִית דַּאֲפִילּוּ אִם לוֹבְשִׁים טַלִּית בַּת אַרְבַּע כְּנָפוֹת אֵין צְרִיכִים לְהַטִּיל בָּהּ צִיצִית מִפְּנֵי דְּמִצְוַת עֲשֵׂה שֶׁהַזְּמַן גְּרָמָא הִיא וְכָל מִצְוַת עֲשֵׂה שֶׁהַזְּמַן גְּרָמָא נָשִׁים וַעֲבָדִים פְּטוּרִים, וְאַף עַל גַּב דְּקַיְמָא לַן בְּשׁוֹפָר וְסֻכָּה וְכַיּוֹצֵא אִם רָצוּ לְקַיֵּם הָרְשׁוּת בְּיָדָם וְיֵשׁ לָהֶם שָׂכָר כְּמִי שֶׁאֵינוֹ מְצֻוֶּה וְעוֹשֶׂה, וְלֵיכָּא מִשּׁוּם כָּל הַפָּטוּר מִן הַדָּבָר וְעוֹשֵׂהוּ נִקְרָא הֶדְיוֹט מִכָּל מָקוֹם בְּמִצְוָה זוֹ מֵחֲזֵי כְּיֻהֲרָא, וְעוֹד יֵשׁ חֲשָׁשׁ נָמֵי מִשּׁוּם "לֹא תִּלְבַּשׁ אִשָּׁה כְּלִי גֶּבֶר", עַל כֵּן בְּצִיצִית צָרִיךְ לִמְנֹעַ אוֹתָם וְכַנִּזְכָּר בַּפּוֹסְקִים, וּבְסֵפֶר הַכַּוָּנוֹת מְפֹרָשׁ טַעַם פְּרָטֵי בְּמִצְוָה זוֹ עַל פִּי הַסּוֹד דְּלָא שָׁיְכָה מִצְוָה זוֹ בְּנָשִׁים:

יד

קָטָן הַיּוֹדֵעַ לְהִתְעַטֵּף צָרִיךְ אָבִיו לִקַּח לוֹ צִיצִית לְחַנְּכוֹ, וְהַאי קָטָן יֵשׁ אוֹמְרִים בֶּן תִּשְׁעָה שָׁנִים, וְיֵשׁ אוֹמְרִים הַפִּקֵּחַ בֶּן שֵׁשׁ וְאֵינוֹ פִּקֵּחַ בֶּן שֶׁבַע צָרִיךְ אָבִיו לִקַּח לוֹ צִיצִית לְחַנְּכוֹ:

טו

לַיְלָה לָאו זְמַן צִיצִית הוּא שֶׁנֶּאֱמַר "וּרְאִיתֶם אוֹתוֹ" פְּרָט לִכְסוּת לַיְלָה, אֲבָל כְּסוּת סוּמָא חַיָּב בְּצִיצִית מִפְּנֵי שֶׁיֶּשְׁנוֹ בִּרְאִיָּה אֵצֶל אֲחֵרִים, וְלָכֵן אֲפִלּוּ אִם יִלְבָּשֶׁנּוּ הַסּוּמָא חַיָּב לְהַטִּיל בּוֹ צִיצִית, וּבְסִפְרֵי הַק' "רַב פְּעָלִים" הֵבֵאתִי בִּתְשׁוּבָתִי שָׁם דִּבְרֵי הָרַב הַגָּאוֹן רַבִּי חַיִּים פָלָאגִ'י זַ"ל בְּ"לֵב חַיִּים" סִימָן ק' שֶׁכָּתַב אַף עַל גַּב דְּקַיְמָא לַן סוּמָא חַיָּב צִיצִית מִכָּל מָקוֹם מֵאַחַר דְּאִיכָּא מַאן דְּאָמַר הַסּוּמָא פָּטוּר יְקַיֵּם מִצְוַת צִיצִית בְּלֹא בְּרָכָה מִשּׁוּם סָפֵק בְּרָכוֹת לְהָקֵל, וְשָׁם הַעֲלֵתִי בִּרְאָיוֹת מֵאַחַר דְּפָשַׁט הַמִּנְהָג שֶׁהַסּוּמָא מְבָרֵךְ יֵשׁ לוֹ לְבָרֵךְ וְאֵין לָחוּשׁ לְסָפֵק בְּרָכוֹת לְהָקֵל, וְגַם הָרַב חַיִּים פָלָאגִ'י זַ"ל עַצְמוֹ כָּתַב כֵּן בְּסֵפֶר "מוֹעֵד לְכָל חַי" סִימָן יוֹ"ד אוֹת מ"ח בְּמָקוֹם שֶׁיֵּשׁ מִנְהָג לָא אָמְרִינַן סָפֵק בְּרָכוֹת לְהָקֵל, וְזֶהוּ מִנְהָג פָּשׁוּט שֶׁהַסּוּמָא מְבָרֵךְ עַל הַטַּלִּית, וְאִם יָבֹא לְהִתְחַסֵּד לִשְׁמֹעַ בְּרָכָה מֵאֲחֵרִים דָּבָר זֶה קָשֶׁה לַעֲשׂוֹתוֹ בְּכָל יוֹם, וְעוֹד נְפַק מִזֶּה פִּקְפּוּק מִצַּד אַחֵר:

טז

מֵחֲמַת דְּאִיכָּא פְּלֻגְתָּא בֵּין הָרַמְבַּ"ם וְהָרֹא"שׁ בִּכְסוּת הַמְיֻחָד לְיוֹם וָלַיְלָה, וּמָרָן זַ"ל לֹא גִּלָּה דַּעְתּוֹ בִּכְסוּת הַמְיֻחָד לַלַּיְלָה וְלוֹבְשׁוֹ בַּיּוֹם. לְכָךְ צָרִיךְ לְהַחְמִיר בְּאִסּוּר תּוֹרָה כְּחֻמְרוֹת שְׁנֵי הַסְּבָרוֹת הַנִּזְכָּרוֹת לְהַטִּיל צִיצִית בִּכְסוּת הַמְיֻחָד לְיוֹם וְלַיְלָה אַף אִם לוֹבְשׁוֹ בַּלַּיְלָה, וְכֵן יַטִּיל לִכְסוּת הַמְיֻחָד לַלַּיְלָה אִם לוֹבְשׁוֹ בַּיּוֹם, אֲבָל לְעִנְיַן הַבְּרָכָה כֵּיוָן דְּסָפֵק בְּרָכוֹת לְהָקֵל לֹא יְבָרֵךְ אֶלָּא עִם לוֹבֵשׁ בַּיּוֹם כְּסוּת הַמְיֻחָד לַיּוֹם, אוֹ כְּשֶׁלּוֹבֵשׁ בַּיּוֹם כְּסוּת הַמְיֻחָד לְיוֹם וָלַיְלָה וְכַנִּזְכָּר בָּאַחֲרוֹנִים זַ"ל:

יז

סְדִינִים וְכָרִים אַף עַל פִּי שֶׁאָדָם יָשֵׁן בָּהֶם לִפְעָמִים בַּיּוֹם, נִקְרָאִים כְּסוּת לַיְלָה דְּעִקָּרָם נַעֲשׂוּ בִּשְׁבִיל לַיְלָה, וּלְמַאן דְּאָמַר כְּסוּת לַיְלָה חַיָּב בְּצִיצִית אִם לוֹבְשׁוֹ בַּיּוֹם, צָרִיךְ לְהַטִּיל בָּהֶם צִיצִית שֶׁהֲרֵי יָשָׁן בָּהֶם בַּבֹּקֶר, וְאַף עַל פִּי כֵן הַמִּנְהָג פָּשׁוּט שֶׁלֹּא לְהַטִּיל בָּהֶם צִיצִית אֲפִלּוּ שֶׁהֵם שֶׁל צֶמֶר, מִפְּנֵי דְּיֵשׁ אוֹמְרִים לֹא חִיְּבָה תּוֹרָה אֶלָּא בְּגָדִים שֶׁאָדָם מִתְכַּסֶּה בָּהֶם דֶּרֶךְ לְבִישָׁה, אֲבָל דֶּרֶךְ הַצָּעָה כְּגוֹן סְדִינִים וְכָרִים לָא מִקְּרֵי כְּסוּת, וּמִכָּל מָקוֹם כֵּיוָן דְּיֵשׁ חוֹלְקִין יַעֲשֶׂה קֶרֶן אַחַת מִן אַרְבַּע קְרָנוֹת עֲגֻלָּה כְּדֵי לְפָטְרָם לְכֻלֵּי עָלְמָא, בַּמֶּה דְבָרִים אֲמוּרִים בִּסְדִינִים שֶׁל צֶמֶר אֲבָל צֶמֶר גֶּפֶן אוֹ מֶשִׁי וְכַיּוֹצֵא אֵין צָרִיךְ לַעֲשׂוֹת בָּהֶם קֶרֶן אַחַת עֲגֻלָּה, דְּיֵשׁ בָּזֶה סְפֵק סְפֵקָא שֶׁמָּא פְּטוּרִים מִן הַתּוֹרָה, וְשֶׁמָּא הֲלָכָה כְּמַאן דְּאָמַר כְּסוּת הַמְיֻחָד לַלַּיְלָה פָּטוּר בַּיּוֹם, וְכֹל שֶׁיֵּשׁ שְׁנֵי סְפֵקוֹת מִן הַתּוֹרָה אַף עַל פִּי שֶׁמִּדְּרַבָּנָן הוּא סָפֵק אֶחָד שַׁפִּיר דָּמֵי כֵּן כָּתַב הַגָּאוֹן "אֵשֶׁל אַבְרָהָם" סִימָן י"ח סְעִיף קָטָן ג, וְנִרְאֶה לִי דְּיֵשׁ בָּזֶה שְׁלֹשָׁה סְפֵקוֹת וְהוּא שֶׁמָּא הֲלָכָה כְּמַאן דְּאָמַר כָּל שֶׁהִיא דֶּרֶךְ הַצָּעָה וְאֵינוֹ דֶּרֶךְ לְבִישָׁה אֵין נִקְרָא כְּסוּת וּפָטוּר, וְעַל כֵּן אֵין לְהַקְפִּיד בִּסְדִינִים שֶׁל שְׁאָר מִינִים, אֲבָל בְּשֶׁל צֶמֶר אַף עַל גַּב דְּיֵשׁ בָּזֶה גַּם כֵּן סְפֵק סְפֵקָא כֵּיוָן דְּאֶפְשָׁר לְתַקֵּן בְּנָקֵל בְּקֶרֶן אַחַת עֲגֻלָּה לָא סָמְכִינַן בְּאִסּוּר תּוֹרָה עַל סְפֵק סְפֵקָא זֶה:

יח

מֻתָּר לְהַטִּיל הַצִּיצִיּוֹת בַּטַּלִּית וּלְקָשְׁרָם בַּלַּיְלָה בְּטַלִּית הַמְיֻחָד לַיּוֹם, אַף עַל גַּב דִּלְהָרַמְבָּ"ם וּדְעִמֵּהּ כְּסוּת יוֹם בַּלַּיְלָה פְּטוּרָה לֵיכָּא בָּזֶה מִשּׁוּם תַּעֲשֶׂה וְלֹא מִן הֶעָשׂוּי, מִשּׁוּם דִּזְמַן מִמֵּילָא קָאָתֵי, וְלָא דָּמֵי לְבַעֲלַת שָׁלֹשׁ דְּאֵינָהּ חַיֶּבֶת וְאֵין רָאוּי לְצִיצִית כְּלָל וְכַנִּזְכָּר בְּ"מִשְׁבְּצוֹת זָהָב" סִימָן י"ח סְעִיף קָטָן א':

יט

הַצִּיצִיּוֹת תַּשְׁמִישֵׁי מִצְוָה הֵם וְאֵין בְּגוּפָן קְדֻשָּׁה, לְפִיכָךְ בִּזְמַן שֶׁקִּבּוּעִין בַּטַּלִּית אָסוּר לְהִשְׁתַּמֵּשׁ בָּהֶם תַּשְׁמִישׁ שֶׁל חוֹל כְּגוֹן לִקְשֹׁר בָּהֶם אֵיזֶה דָּבָר וְכַיּוֹצֵא, גַּם יִזָּהֵר שֶׁלֹּא יִהְיוּ נִגְרָרִין עַל גַּבֵּי קַרְקַע מִשּׁוּם בִּזְיוֹן מִצְוָה, וְאַחַר שֶׁנִּפְסְקוּ מִן הַטַּלִּית יָכוֹל לְזָרְקָן וְרַק לֹא יִזְרְקֵם לָאַשְׁפָּה אוֹ לְמָקוֹם מְגֻנֶּה, וְיֵשׁ מַחְמִירִין לְגָנְזָן וְתָבֹא עֲלֵיהֶם בְּרָכָה:

כ

טַלִּיתוֹת שֶׁל מִצְוָה שֶׁבָּלוּ, אָסוּר לְקַנֵּחַ בָּהֶם אוֹ לְהִשְׁתַּמֵּשׁ בָּהֶם תַּשְׁמִישׁ מְגֻנֶּה אֶלָּא זוֹרְקָן וְהֵם כָּלִים מֵאֲלֵיהֶם, וְאוֹתָם שֶׁנּוֹהֲגִין לְזָרְקָן בְּבֵית הַכְּנֶסֶת וְהַשַּׁמָּשִׁים מְקַנְּחִין בָּהֶם נֵרוֹת בֵּית הַכְּנֶסֶת, וְגַם יְדֵיהֶם הַמְלֻכְלָכִים בְּשֶׁמֶן, זֶה נִקְרָא תַּשְׁמִישׁ מְגֻנֶּה וְאָסוּר:

כא

מֻתָּר לַעֲשׂוֹת מִטַּלִּית גָּדוֹל טַלִּית קָטָן, כֵּן הֶעֱלָה הָרַב "בֵּית יְהוּדָה" סִימָן כ"ט, אֲבָל הָרַב "לֵב חַיִּים" זַ"ל סִימָן צ"ב פִּקְפֵּק בָּזֶה, מִשּׁוּם דְּעַל פִּי הַסּוֹד טַלִּית גָּדוֹל יֵשׁ בּוֹ מַעֲלָה רַבָּה עַל טַלִּית קָטָן וְהָאֱמֶת אִתּוֹ בָּזֶה, מִיהוּ אִם נִתַּבְּלָה מִקְצָתוֹ וְאֵין רָאוּי לְלָבְשׁוֹ יוּכַל לַעֲשׂוֹת מֵחֵלֶק הֶחָזָק שֶׁבּוֹ טַלִּית קָטָן וְאֵין חֲשָׁשׁ בָּזֶה לְכֻלֵּי עָלְמָא:
